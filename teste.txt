double maior (double a, double b){
  int m;
  
  if (a > b) {m = a;}
  else {m = b;}
  return m;
}

int fat (int n)
{
  int f;
  
  f = 0;
  while (n > 0)
  { 
    f = f * n;
    n = n - 1;
  }
}
